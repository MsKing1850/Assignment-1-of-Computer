Q-1: Write a program to print numbers from 1 to 10 using
(a) For Loop
(b) While Loop
(c) Do-While Loop

      
Input:- 

#include <stdio.h>

int main() {
  int i;
  printf("Numbers from 1 to 10 using For Loop:\n");
  for ( i = 1; i <= 10; i++) {
      printf("%d ", i);
  }
  printf("\n\n");

  printf("Numbers from 1 to 10 using While Loop:\n");
  i = 1;
  while (i <= 10) {
      printf("%d ", i);
      i++;
  }
  printf("\n\n");


  printf("Numbers from 1 to 10 using Do-While Loop:\n");
  i = 1;
  do {
      printf("%d ", i);
      i++;
  } while (i <= 10);
  printf("\n");

}

            printf("Factorial is not defined for negative numbers.\n");

Q-8: Write a program to print all prime numbers between 1 and 100 using nested loops.

      
Input:- 

      #include <stdio.h>

      int main() {
        int num, i, isPrime;
    
        printf("Prime numbers between 1 and 100 are:\n");
    
       
        for (num = 2; num <= 100; num++) {
            isPrime = 1;  
    
            for (i = 2; i * i <= num; i++) {
                if (num % i == 0) {
                    isPrime = 0;  
                    break;        
                }
            }
    
            if (isPrime == 1) {
                printf("%d ", num);
            }
        }
    
        printf("\n");
       
    }

Q-9: Write a program to display the first 10 terms of the Fibonacci sequence using a for loop.

      
Input:- 

      #include <stdio.h>
      int main() {
        int terms = 10;
        int first = 0, second = 1, next;
    
        printf("First 10 terms of the Fibonacci sequence:\n");
    
        printf("%d %d ", first, second);
    
        for (int i = 3; i <= terms; i++) {
            next = first + second;  
            printf("%d ", next);   
            first = second;       
            second = next;         
        }
    
        printf("\n");
      
    }
